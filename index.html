import os import webbrowser # Definindo os dados do jogador jogador = { "nome": "Neymar Jr", "posicao": "Atacante / Meio-Campista", "clube": "Al-Hilal", "overall": 89, "imagem": "jogador.jpg" # Nome do arquivo que está na sua pasta } def mostrar_status(): print("\n=== FICHA DO JOGADOR ===") print(f"Nome: {jogador['nome']}") print(f"Posição: {jogador['posicao']}") print(f"Clube Atual: {jogador['clube']}") print(f"Overall: {jogador['overall']}") print("========================\n") def abrir_imagem(): # Verifica se a imagem realmente existe na pasta antes de abrir if os.path.exists(jogador["imagem"]): print(f"Abrindo a imagem de {jogador['nome']}...") webbrowser.open(jogador["imagem"]) else: print(f"Erro: O arquivo '{jogador['imagem']}' não foi encontrado na pasta!") # Loop de comandos while True: print("Comandos disponíveis: [1] Ver Status | [2] Ver Imagem | [3] Sair") comando = input("Digite o número do comando: ") if comando == "1": mostrar_status() elif comando == "2": abrir_imagem() elif comando == "3": print("Saindo do programa... Valeu!") break else: print("Comando inválido! Tente novamente.\n")
